## 目录结构

```
.
├── app.js
├── app.json
├── app.wxss
├── pages 
│   ├── index //主界面文件夹
│   │   ├── index.js
│   │   ├── index.json
│   │   ├── index.wxml
│   │   └── index.wxss
│   ├── management 
│   │   ├── .DS_Store
│   │   └── change //变更管理文件夹
│   │       ├── .DS_Store
│   │       ├── launch //变更管理-发起文件夹
│   │       ├── submission //变更管理-抄送文件夹
│   │       ├── treated  //变更管理-已处理文件夹
│   │       └── untreated  //变更管理-未处理文件夹
│   ├── upcome //待办事项文件夹
│   └── user //个人中心文件夹   
├── project.config.json
├── sitemap.json
├── static //图片资源
│   ├── image
│   └── tab
└── utils
    └── util.js
```
